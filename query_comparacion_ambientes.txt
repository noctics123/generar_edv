-- Query para PRODUCCION (el que ya ejecutaste)
SELECT nomgrupo, nomtablaconcepto, COUNT(*) as num_columnas
FROM catalog_lhcl_prod_bcp.bcp_ddv_matrizvariables_v.mm_parametrogrupoconcepto
WHERE nomgrupo IN ('P1', 'P2')
GROUP BY nomgrupo, nomtablaconcepto
ORDER BY nomgrupo, nomtablaconcepto;

-- Query para DESARROLLO (ejecuta esta si tienes acceso)
SELECT nomgrupo, nomtablaconcepto, COUNT(*) as num_columnas
FROM catalog_lhcl_desa_bcp.bcp_ddv_matrizvariables.mm_parametrogrupoconcepto
WHERE nomgrupo IN ('P1', 'P2')
GROUP BY nomgrupo, nomtablaconcepto
ORDER BY nomgrupo, nomtablaconcepto;

-- Para ver el contenido completo de nomtablaconcepto sin truncar en PROD
SELECT DISTINCT nomgrupo, nomtablaconcepto
FROM catalog_lhcl_prod_bcp.bcp_ddv_matrizvariables_v.mm_parametrogrupoconcepto
WHERE nomgrupo = 'P2'
ORDER BY nomtablaconcepto;
